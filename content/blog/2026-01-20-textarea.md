import base64

import zlib

def decode\_textarea\_my\_url(url):

\# 1. Extract the hash fragment (the part after #)

if '#' not in url:

return "Error: No hash fragment found in URL."

base64\_string = url.split('#')\[1\]

\# 2. Handle Base64 Padding

\# Base64URL strings often omit the '=' padding at the end,

\# but Python's decoder requires it.

padding = 4 - len(base64\_string) % 4

if padding != 4:

base64\_string += '=' \* padding

\# 3. Decode Base64URL to Bytes

try:

compressed\_data = base64.urlsafe\_b64decode(base64\_string)

except Exception as e:

return f"Error decoding Base64: {e}"

\# 4. Decompress Deflate-Raw

\# -15 is the window bits value for raw deflate (no headers/checksums),

\# which matches the 'deflate-raw' used by the website.

try:

decompressed\_data = zlib.decompress(compressed\_data, -15)

except zlib.error:

return "Error: Could not decompress data. It might be corrupted or use a different format."

\# 5. Decode Bytes to String

\# The data often contains a NULL byte (\\x00) separating text content from style JSON.

\# We only care about the text content (first part).

try:

full\_string = decompressed\_data.decode('utf-8')

content\_text = full\_string.split('\\x00')\[0\]

return content\_text

except UnicodeDecodeError:

return "Error: Could not decode bytes to UTF-8 string."
